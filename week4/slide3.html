• scope: by default, local
  • global keyword allows outside access, which is generally bad
    •  <?php
$var = 5;
function printVar() { print $var; }
//this will print nothing (and probably give an error);

function printVar() {
  global $var;
  print $var;
}
//this will print "5";
?>
  • return multiple values:
    •  
function returnMultiples($r) {
  $double = $r*2;
  $triple = $r*3;
  return array($double, $triple);
}
list($d, $t) = returnMultiples(5);
print "Double:$d --- Triple:$t<br />";
